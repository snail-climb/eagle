Hadoop集群部署文档

本次部署的环境及版本信息

  1. CentOS 8 Stream 最小化安装
  2. 网络模式：NAT模式
  3. IP地址：192.168.10.1
  4. 子网：192.168.10.0
  5. 子网掩码：255.255.255.0
  6. 网关：192.168.10.2
  7. JDK版本：8u391
  8. Hadoop版本：3.3.6

1. 工具安装

dnf install -y lrzsz
dnf install -y nc
dnf install -y net-tools
dnf install -y rsync
dnf install -y vim

2.2 修改主机名称

# vim /etc/hostname

2.3 添加映射关系

# vim /etc/hosts
192.168.10.100 gemini
192.168.10.101 hadoop101
192.168.10.102 hadoop102
192.168.10.103 hadoop103

2.4 关闭防火墙

systemctl stop firewalld
systemctl disable firewalld.service

2.5 配置非root用户

useradd hadoop
passwd hadoop

# vim /etc/sudoers
root    ALL=(ALL)       ALL
%wheel  ALL=(ALL)       ALL
hadoop  ALL=(ALL)       NOPASSWD:ALL

3. 软件安装(非root用户操作)

su hadoop
sudo mkdir /opt/module /opt/software
sudo chown hadoop:hadoop /opt/module/ /opt/software/
tar -zxvf /opt/software/jdk-8u391-linux-x64.tar.gz -C /opt/module/
tar -zxvf /opt/software/hadoop-3.3.6.tar.gz -C /opt/module/

# vim /etc/profile.d/my_env.sh
export JAVA_HOME=/opt/module/jdk1.8.0_391
export PATH=$PATH:$JAVA_HOME/bin

export HADOOP_HOME=/opt/module/hadoop-3.3.6
export PATH=$PATH:$HADOOP_HOME/bin
export PATH=$PATH:$HADOOP_HOME/sbin

source /etc/profile

4. ssh免密登录

# root和非root用户各执行一遍
ssh-keygen -t rsa
ssh-copy-id hadoop101
ssh-copy-id hadoop102
ssh-copy-id hadoop103

5. 内容分发

# Secure Copy(同步用scp -r; 修改用rsync -av)
# 主动推
scp -r jdk1.8.0_391/ hadoop@hadoop102:/opt/module
# 主动拉
scp -r hadoop@hadoop101:/opt/module/hadoop-3.3.6 ./
# 指定发送和接收方
scp -r hadoop@hadoop101:/opt/module/* hadoop@hadoop103:/opt/module/

# 编写xsync(对rsync的封装)
cd /home/hadoop
mkdir bin
cd bin
vim xsync
chmod 777 xsync

#!/bin/bash

if [ $# -lt 1 ]
then
    echo Not Enough Arguements!
    exit;
fi

for host in hadoop101 hadoop102 hadoop103
do
    echo ====================  $host  ====================
    for file in $@
    do
        if [ -e $file ]
        then
            pdir=$(cd -P $(dirname $file); pwd)
            fname=$(basename $file)
            ssh $host "mkdir -p $pdir"
            rsync -av $pdir/$fname $host:$pdir
        else
            echo $file does not exist!
        fi
    done
done

2. 集群配置

2.1 核心配置文件

<!-- core-site.xml -->
<configuration>
    <property>
        <name>fs.defaultFS</name>
        <value>hdfs://hadoop101:8020</value>
    </property>
    <property>
        <name>hadoop.tmp.dir</name>
        <value>/opt/module/hadoop-3.3.6/tmp</value>
    </property>
    <property>
        <name>hadoop.http.staticuser.user</name>
        <value>hadoop</value>
    </property>
</configuration>

2.2 HDFS配置文件

<!-- hdfs-site.xml -->
<configuration>
    <property>
        <name>dfs.namenode.http-address</name>
        <value>hadoop101:9870</value>
    </property>
    <property>
        <name>dfs.namenode.secondary.http-address</name>
        <value>hadoop102:9868</value>
    </property>
</configuration>

2.3 MapReduce配置文件

<!-- mapred-site.xml -->
<configuration>
    <property>
        <name>mapreduce.framework.name</name>
        <value>yarn</value>
    </property>
</configuration>

2.4 YARN配置文件

<!-- yarn-site.xml -->
<configuration>
    <property>
        <name>yarn.resourcemanager.hostname</name>
        <value>hadoop103</value>
    </property>
    <property>
        <name>yarn.resourcemanager.webapp.address</name>
        <value>${yarn.resourcemanager.hostname}:8088</value>
    </property>
    <property>
        <name>yarn.nodemanager.aux-services</name>
        <value>mapreduce_shuffle</value>
    </property>
</configuration>

2.5 hadoop-env.sh

# hadoop-env.sh
export JAVA_HOME=/opt/module/jdk1.8.0_391

3. 集群启动

3.1 集群初始化(首次启动)

3.2 集群启动

# hadoop101
sbin/start-dfs.sh
# hadoop103
sbin/start-yarn.sh
